## Install

```bash
# Install vpopup for Vue 3 project
npm i @a0znpm/vpopup -S
```

## Quickstart

```js
import { createApp } from 'vue';
import VPopup from '@a0znpm/vpopup';

const app = createApp();
app.use(VPopup);
```
